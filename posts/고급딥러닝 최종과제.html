train_labels = train_DNN.pop('isDM')

test_labels = test_DNN.pop('isDM')

def norm(x):
  return (x - train_DNN_stats['mean']) / train_DNN_stats['std']

normed_train_DNN = norm(train_DNN)

normed_test_DNN= norm(test_DNN)
def build_model():
  model = keras.Sequential([

    layers.Dense(128, activation='relu', input_shape=[len(normed_train_DNN.keys())]),
    layers.Dense(64, activation='relu'),
    layers.Dense(32, activation='relu'),
    layers.Dense(1, activation='sigmoid')])

  optimizer = tf.keras.optimizers.Adam(0.0001)
  model.compile(loss='binary_crossentropy',  # 분류 문제에 적합한 손실 함수 선택
                  optimizer=optimizer,
                  metrics=['accuracy'])

  return model

def plot_history(history):
  hist = pd.DataFrame(history.history)
  hist['epoch'] = history.epoch

  plt.figure(figsize=(8,12))

  plt.subplot(2,1,1)
  plt.xlabel('Epoch')
  plt.ylabel('Loss')
  plt.plot(hist['epoch'], hist['loss'], label = 'Train loss')
    
  plt.plot(hist['epoch'], hist['val_loss'], label = 'Validation loss')
  plt.title('Loss')
  plt.legend()
 
    
  plt.subplot(2,1,2)
  plt.xlabel('Epoch')
  plt.ylabel('Accuracy')
  plt.plot(hist['epoch'], hist['accuracy'], label = 'Train accuracy')

  plt.plot(hist['epoch'],hist ['val_accuracy'], label = 'Validation val_accuracy')

  plt.title('Accuracy')
  plt.legend()
  plt.show()
